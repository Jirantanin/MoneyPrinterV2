```python
def render(self) -> None:
    """Render 14 scene clips and concatenate via FFmpeg into final.mp4.

    Pass 1: For each scene, combine scene_NN.png + scene_NN.wav into
    scene_NN.mp4 using FFmpeg zoompan Ken Burns effect. Even scenes
    zoom in (z='1+0.1*on/{frames}'), odd scenes zoom out
    (z='1.1-0.1*on/{frames}'). Skips clips that already exist.

    Pass 2: Writes concat_list.txt and runs FFmpeg concat demuxer
    to produce final.mp4 via stream-copy (no re-encode).

    Raises:
        ValueError: If episode_dir is not set.
        FileNotFoundError: If scene_NN.png or scene_NN.wav is missing.
        subprocess.CalledProcessError: If FFmpeg or ffprobe exits non-zero.
    """
    if not self.episode_dir:
        raise ValueError(
            "episode_dir is not set. Call generate_script() before render()."
        )

    FPS = 25

    # --- Pass 1: render per-scene clips ---
    for i in range(14):
        scene_num = str(i).zfill(2)
        png_path = os.path.join(self.episode_dir, f"scene_{scene_num}.png")
        wav_path = os.path.join(self.episode_dir, f"scene_{scene_num}.wav")
        clip_path = os.path.join(self.episode_dir, f"scene_{scene_num}.mp4")

        # Resumability: skip if clip already exists
        if os.path.exists(clip_path):
            print(f"Skipping scene {i + 1}/14 (clip already exists).")
            continue

        # Guard: assets must exist before rendering
        if not os.path.exists(png_path):
            raise FileNotFoundError(
                f"scene_{scene_num}.png not found at {png_path}. "
                "Run generate_assets() first."
            )
        if not os.path.exists(wav_path):
            raise FileNotFoundError(
                f"scene_{scene_num}.wav not found at {wav_path}. "
                "Run generate_assets() first."
            )

        # Get audio duration via ffprobe
        result = subprocess.run(
            [
                "ffprobe", "-v", "error",
                "-show_entries", "format=duration",
                "-of", "default=noprint_wrappers=1:nokey=1",
                wav_path,
            ],
            capture_output=True, text=True, check=True,
        )
        duration = float(result.stdout.strip())
        frames = math.ceil(duration * FPS)

        # Choose zoom direction: even = zoom-in, odd = zoom-out
        if i % 2 == 0:
            z_expr = f"1+0.1*on/{frames}"
        else:
            z_expr = f"1.1-0.1*on/{frames}"

        # Build filter_complex string.
        # scale+pad is MANDATORY: source PNGs are 768x1344; zoompan requires
        # input dimensions to exactly match s=1080x1920 (Pitfall 2 in RESEARCH.md).
        filter_str = (
            f"[0:v]scale=1080:1920:force_original_aspect_ratio=decrease,"
            f"pad=1080:1920:(ow-iw)/2:(oh-ih)/2,"
            f"zoompan=z='{z_expr}':x='iw/2-(iw/zoom/2)':y='ih/2-(ih/zoom/2)':"
            f"d={frames}:s=1080x1920:fps=25[v]"
        )

        cmd = [
            "ffmpeg", "-y",
            "-loop", "1", "-i", png_path,
            "-i", wav_path,
            "-filter_complex", filter_str,
            "-map", "[v]", "-map", "1:a",
            "-c:v", "libx264", "-preset", "fast",
            "-pix_fmt", "yuv420p",
            "-c:a", "aac", "-b:a", "192k",
            "-t", str(duration),
            clip_path,
        ]

        print(f"Rendering scene {i + 1}/14...")
        try:
            subprocess.run(cmd, capture_output=True, text=True, check=True)
        except subprocess.CalledProcessError as e:
            raise subprocess.CalledProcessError(
                e.returncode, e.cmd,
                output=e.output,
                stderr=e.stderr,
            ) from e

    # --- Pass 2: concatenate all clips into final.mp4 ---
    # Forward-slash paths are REQUIRED in concat_list.txt on Windows (Pitfall 3).
    concat_path = os.path.join(self.episode_dir, "concat_list.txt")
    with open(concat_path, "w", encoding="utf-8") as f:
        for i in range(14):
            scene_num = str(i).zfill(2)
            clip_path = os.path.join(self.episode_dir, f"scene_{scene_num}.mp4")
            clip_fwd = clip_path.replace(chr(92), "/")
            f.write(f"file '{clip_fwd}'\n")

    final_path = os.path.join(self.episode_dir, "final.mp4")
    cmd_concat = [
        "ffmpeg", "-y",
        "-f", "concat", "-safe", "0",
        "-i", concat_path,
        "-c", "copy",
        final_path,
    ]
    print("Concatenating clips into final.mp4...")
    subprocess.run(cmd_concat, capture_output=True, text=True, check=True)

    # Post-render sanity check
    assert os.path.exists(final_path), f"final.mp4 not found at {final_path}"
    assert os.path.getsize(final_path) > 1_000_000, (
        f"final.mp4 is suspiciously small ({os.path.getsize(final_path)} bytes) — "
        "check FFmpeg output."
    )

    print(f"Final video: {final_path}")
```

Critical implementation notes (from RESEARCH.md pitfalls):
- Use `math.ceil(duration * FPS)` for frames — NOT `int()`. Floor by 1 frame causes zoom reset mid-clip (Pitfall 1).
- Use `-t str(duration)` — NOT `-shortest`. `-shortest` can cut the last second of audio (Pitfall 4).
- Use `-filter_complex` with `[0:v]...[v]` + explicit `-map "[v]"` + `-map "1:a"` — required when mixing `-loop 1` still image with audio input.
- Pass `filter_str` as a single list element — no shell escaping needed. Single quotes inside the zoompan z= expression are FFmpeg filter syntax, not shell quotes.
- The `scale+pad` chain before zoompan is mandatory — source PNGs are 768x1344, zoompan requires input to match `s=1080x1920` or exits with "Invalid argument" and zero-byte output (Pitfall 2).
- Write forward slashes in concat_list.txt using `clip_path.replace(chr(92), "/")` — FFmpeg concat parser rejects backslashes on Windows (Pitfall 3).
- Use `check=True` on all subprocess.run calls — raises CalledProcessError on non-zero exit.
- Do NOT add `shell=True` — FFmpeg is a native .exe, not a .cmd wrapper.
